    <h1>🎲 Random Trái Ác Quỷ 🎲</h1>
    <button onclick="showRandomFruit()">Nhận Ngay</button>
    <p id="result">🍏 Nhấn nút để random!</p>

    <script>
        // Danh sách trái ác quỷ + tỷ lệ xuất hiện (Tổng = 100%)
        const bloxFruits = [
            { name: "Leopard 🐆 (Mythical)", chance: 5 },   // 5%
            { name: "Dragon 🐉 (Mythical)", chance: 1 },    // 1%
            { name: "Dough 🍩 (Mythical)", chance: 15 },    // 15%
            { name: "Venom 🐍 (Mythical)", chance: 10 },    // 10%
            { name: "Shadow 👤 (Mythical)", chance: 10 },   // 10%
            { name: "Spirit 👻 (Mythical)", chance: 10 },   // 10%
            { name: "Control ☯️ (Mythical)", chance: 5 },   // 5%
            { name: "Gravity 🌍 (Legendary)", chance: 7 },  // 7%
            { name: "Portal 🚪 (Legendary)", chance: 8 },   // 8%
            { name: "Rumble ⚡ (Legendary)", chance: 10 },   // 10%
            { name: "Blizzard ❄️ (Legendary)", chance: 10 }, // 10%
            { name: "Yeti ☃ (Mythical)", chance: 4 },      // 4% (Đã đổi emoji ☃)
            { name: "Kitsune 🦊 (Mythical)", chance: 5 }    // 5%
        ];

        function showRandomFruit() {
            let randomNumber = Math.random() * 100; // Tạo số ngẫu nhiên từ 0 - 100
            let cumulativeChance = 0;

            for (let fruit of bloxFruits) {
                cumulativeChance += fruit.chance; // Cộng dồn xác suất
                if (randomNumber < cumulativeChance) {
                    document.getElementById("result").innerText = `🔥 Bạn nhận được: ${fruit.name}!`;
                    return;
                }
            }
        }
    </script>
